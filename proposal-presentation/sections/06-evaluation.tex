\section{Evaluation Framework}

\begin{frame}\frametitle{\secname}\framesubtitle{Human Evaluation}
    \begin{itemize}
        \item<1-> \textbf{Participants}: Semi-professional audio engineers and producers
        \item<2-> \textbf{Evaluation Criteria}:
        \begin{itemize}
            \item<3-> \textbf{Effectiveness}: How well does the advice address the mixing challenge?
            \item<4-> \textbf{Actionability}: How clear and implementable is the advice?
            \item<5-> \textbf{Adherence to Conventions}: How well does the advice follow established mixing practices?
        \end{itemize}
        \item<6-> \textbf{Methodology}: [To be determined - considering multiple evaluation approaches]
    \end{itemize}
\end{frame}
