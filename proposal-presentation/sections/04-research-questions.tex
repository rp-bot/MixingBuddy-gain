\section{Research Questions}

\begin{frame}\frametitle{\secname}\framesubtitle{primary}
\textbf{Primary Research Question}
\begin{itemize}
    \item To what extent can an \textbf{Audio-Language Model} learn the \textbf{relative gain} relationships among multitrack stems and generate musically effective gain-balancing \textbf{advice}?
\end{itemize}
\end{frame}

\begin{frame}\frametitle{\secname}\framesubtitle{secondary}
\textbf{Secondary Questions}
\begin{itemize}
    \item<1-> \textbf{Model Understanding}: What model architecture best represents and reasons about the input mix for learning relative gain relationships?
    \item<2-> \textbf{Mixing Conventions}: To what extent does the model's advice reflect established mixing conventions?
    \item<3-> \textbf{Communication}: How effectively does the model communicate its advice in a way that is clear, actionable, and ``correct''?
    \item<4-> \textbf{Usefulness}: How do audio engineers and producers evaluate the effectiveness, musicality, and real-world usefulness of the advice?
\end{itemize}
\end{frame}
