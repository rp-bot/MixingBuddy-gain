\section{Research Questions}

\begin{frame}\frametitle{\secname}\framesubtitle{primary}
\textbf{Primary Research Question}
\begin{itemize}
    \item To what extent can an Audio-Language Model, conditioned on an anchor track, learn the relative gain relationships among multitrack stems and generate musically effective gain-balancing advice?
\end{itemize}
\end{frame}

\begin{frame}\frametitle{\secname}\framesubtitle{secondary}
\textbf{Secondary Questions}
\begin{itemize}
    \item<1-> \textbf{Model Understanding}: What model architecture best represents and reasons about multitrack stems and anchor tracks for learning relative gain relationships?
    \item<2-> \textbf{Mixing Conventions \& Genre Awareness}: To what extent does the model's advice reflect established mixing conventions, and how does its performance vary across different musical genres?
    \item<3-> \textbf{Communication \& Actionability}: How effectively does the model communicate its advice in a way that is clear, actionable, and distinct from simply being "correct"?
    \item<4-> \textbf{Human Evaluation \& Usefulness}: How do audio engineers and producers evaluate the effectiveness, musicality, and real-world usefulness of the advice in their workflows?
\end{itemize}
\end{frame}
