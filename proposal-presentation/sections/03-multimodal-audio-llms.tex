    \item<1-> \textbf{Direct Tokenization (Unified Approach)}: converts raw audio into discrete tokens via audio codecs. the LLM vocabulary is extended to include audio tokens \cite{Rubenstein_Asawaroengchai_Nguyen_Bapna_Borsos_Quitry_Chen_Badawy_Han_Kharitonov_etal_2023, Du_Wang_Chen_Chu_Gao_Li_Hu_Zhou_Xu_Ma_etal_2024, Zhang_Li_Zhang_Zhan_Wang_Zhou_Qiu_2023}.
    \item<2-> \textbf{Feature Extraction (Cascade Approach)}: uses audio-specific encoders/decoders with the LLM as a central backbone (e.g., LTU) \cite{Liu_Hussain_Wu_Sun_Shan_2024, Gong_Luo_Liu_Karlinsky_Glass_2024}.
\end{itemize}
